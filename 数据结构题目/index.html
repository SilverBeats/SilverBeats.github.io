public SingleLinearNode reverseLinear(SingleLinearNode linear) {
  SingleLinearNode head = new SingleLinearNode(), temp, lp;
  lp = linear;
  while (lp != null) {
    temp = lp.getNext();
    lp.setNext(head.getNext());
    head.setNext(lp);
    lp = temp;
  }
  return head.getNext();
}

1.2 将两个有序的单链表合并，合并后结果仍然有序

将两个有序的单链表合并，合并后结果仍然有序（假定两个链表不含头结点并非减）

public SingleLinearNode combineTwoLinear(SingleLinearNode linear1, SingleLinearNode linear2) {
  SingleLinearNode p1 = linear1, p2 = linear2;
  SingleLinearNode head = new SingleLinearNode(), hp = head, temp;
  while (p1 != null && p2 != null) {
    if (p1.getElemData() < p2.getElemData()) {
      hp.setNext(p1);
      hp = p1;
      p1 = p1.getNext();
    } else {
      hp.setNext(p2);
      hp = p2;
      p2 = p2.getNext();
    }
    hp.setNext(null);
  }
  // 如果某个链表还有节点
  if (p1 != null) hp.setNext(p1);
  if (p2 != null) hp.setNext(p2);
  return head.getNext();
}

1.4 一个数组正数负数混合，以最快的速度将所有的负数挪到整个数组的最前面

一个数组，其中有正数和负数，以最快的速度将所有的负数挪到整个数组的最前面

public void negativeFrontPositive(int[] arr) {
  if (arr.length <= 1) return;
  // left表示负数存储的下标,right表示正数的位置
  int left = 0, right = arr.length - 1;
  while (left < right) {
    while (arr[left] < 0) ++left;
    while (arr[right] > 0) --right;
    // 经过两个while,left的位置是正数,right的位置是负数,进行交换
    swap(arr, left, right);
    ++left;
    --right;
  }
}

1.5 一个递增有序数组，以最快的方式找到某个值，找到则将该值与后继进行交换，找不到则插入

一个递增有序数组，以最快的方式找到某个值，找到则将该值与后继进行交换，找不到则插入

public int[] findSwapOrInsert(int[] arr, int target) {
  if (arr.length == 0) return new int[]{target};
  // 二分查找
  int left = 0, right = arr.length - 1, mid;
  while (left < right) {
    mid = (left + right) >> 1;
    if (arr[mid] > target) right = mid - 1;
    else if (arr[mid] < target) left = mid + 1;
    else {
      // 如果找到了
      swap(arr, mid, mid + 1);
      return arr;
    }
  }
  // 如果没找到
  int i = Math.max(right, 0);
  for (; i < arr.length; i++) {
    if (target < arr[i]) break;
  }
  // 此时i就是插入位置
  int[] newArr = new int[arr.length + 1];
  System.arraycopy(arr, 0, newArr, 0, i);
  newArr[i] = target;
  System.arraycopy(arr, i, newArr, i + 1, arr.length - i);
  return newArr;
}

1.6 判断某单链表前n个字符是否是中心对称

判断某单链表前n个字符是否是中心对称

public boolean isCenterSymmetry(SingleLinearNode linear, int len) {
  // 先统计linear有多少个节点
  int count = 0;
  SingleLinearNode lp = linear;
  while (lp != null) {
    ++count;
    lp = lp.getNext();
  }
  // 可能会出现链表4个节点,让判断前5个乃至更多是否中心对称
  int realN = Math.min(len, count);
  // 创建一个数组用来存储linear前realN个关键字
  int[] arr = new int[realN];
  for (int i = 0; i < realN; i++) {
    arr[i] = linear.getElemData();
    linear = linear.getNext();
  }
  // 定义左右指针
  int left = 0, right = realN - 1;
  // 判断是否中心对称
  while (left < right) {
    if (arr[left++] != arr[right--]) return false;
  }
  return true;
}

1.7 删除不带头结点的单链表，begin位置后的len个节点

删除不带头结点的单链表，begin位置后的len个节点

  • begin = 0：从单链表第一个节点开始删
  • begin = 1：从单链表第二个位置开始删除，第一个节点保留
  • ……
  • 起始位置超出链表长度：报错
  • 能删除的节点数量不足len个：有多少删多少
public SingleLinearNode deleteNodeFrom(SingleLinearNode linear, int begin, int len) {
  if (begin < 0) throw new RuntimeException("deleteNodeFrom参数错误,begin需要≥0");
  if (len < 0) throw new RuntimeException("deleteNodeFrom参数错误,len需要≥0");
  if (linear == null) return null;
  // 令传递过来的单链表配上一个头结点
  SingleLinearNode head = new SingleLinearNode(), hp = head;
  head.setNext(linear);
  int i = 0;
  while(i++ < begin && hp != null) hp = hp.getNext();
  if(hp == null) throw new RuntimeException("起始位置超出链表长度");
  // 此时,从hp后面的节点开始计数len个要删除
  SingleLinearNode temp = hp;
  for(i = 1; i <= len && temp != null; i++) temp = temp.getNext();
  // 此时temp指向被删除的最后一个节点
  if(temp != null) hp.setNext(temp.getNext());
  // 说明能删的不够len个
  else hp.setNext(null);
  return head.getNext();
}

public SingleLinearNode q7(SingleLinearNode linear1, int begin, int len, SingleLinearNode linear2, int insertPos) {
  if(insertPos < 0) throw new RuntimeException("linear2插入位置需要≥0");
  // 先对linear1进行处理,进行删除
  linear1 = deleteNodeFrom(linear1, begin, len);
  // 判断经过删除后,linear1还有没有可以插入的节点
  if(linear1 != null) {
    // p定位到linear1的最后一个节点
    SingleLinearNode p = linear1;
    while(p.getNext() != null) p = p.getNext();
    if(insertPos > 0) {
      // 找到linear的第j个位置的前一个位置
      SingleLinearNode prePos = getPos(linear2, insertPos - 1);
      if(prePos != null) {
        p.setNext(prePos.getNext());
        prePos.setNext(linear1);
      } else {
        // 如果定位失败,则将linear1剩下的节点插入到linear2的最后
        // 此时p指向linear2的最后一个节点
        p = linear2;
        while(p.getNext() != null) p = p.getNext();
        // linear1剩余节点插入linear2最后
        p.setNext(linear1);
      }
    } else {
      // 若j=0
      p.setNext(linear2);
      return linear1;
    }
  }
  return linear2;
}

2.1 给定入栈序列和某个出栈序列，判断该出栈序列是否合法

给定入栈序列和某个出栈序列，判断该出栈序列是否合法

/**
	* @param inStackSeq: 入栈顺序
	* @param outStackSeq: 出栈顺序
	* @return: boolean
	*/
public static boolean isRightOutStackSeq(String inStackSeq, String outStackSeq) {
  if(inStackSeq.length() != outStackSeq.length()) return false;
  // 入栈序列和出栈序列转为数组,方便处理
  char[] inStackSeqChars = inStackSeq.toCharArray();
  char[] outStackSeqChars = outStackSeq.toCharArray();
  // 声明一个栈
  SequenceStack<Character> stack = new SequenceStack<>(inStackSeqChars.length);
  // 声明一个指针i,指向出栈序列,这里不借助队列实现
  // 指针j指向入栈序列
  int i = 0, j = 0;
  while(j < inStackSeqChars.length) {
    stack.push(inStackSeqChars[j++]);
    // 每次入栈和出栈顺序队列第一个比较,如果相等
    while(!stack.isEmpty() && stack.peek() == outStackSeqChars[i]) {
      i++;
      stack.pop();
    }
  }
  return i >= outStackSeqChars.length && stack.isEmpty();
}

2.2 前缀、中缀、后缀表达式

/**	
	* @param char1: 第一个运算符
	* @param char2: 第二个运算符
	* @return: int
	* @description: 返回0,优先级一致;正数,前面优先级高;负数,前面优先级低
	*/
private int cmp(char char1, char char2) {
  Map<Character, Integer> map = new HashMap<>();
  char[] chars = {'/', '*', '%'};
  for (char item : chars) map.put(item, 3);
  chars = new char[]{'+', '-'};
  for (char item : chars) map.put(item, 4);
  chars = new char[]{'^'};
  for (char item : chars) map.put(item, 9);
  return map.get(char2) - map.get(char1);
}
/**
	* @param op: 操作符
	* @return: boolean
	* @description: 判断是否是操作符
	*/
private boolean isOp(char op) {
  return op == '*' || op == '+' || op == '/' || op == '-' || op == '%' || op == '^';
}

2.2.1 中缀——>前缀表达式

遵循右优先，倘若一个中缀表达式右边的运算符能够先运算，该右边的运算符优先级高

规则：

  • 从右向左遍历
  • 操作数前插
  • 操作符，若栈空入栈；不空，则将栈中大于当前运算符优先级出栈进行前插,或遇到右括号停止。之后再将当前运算符压栈
  • 右括号入栈，遇到左括号不断弹出操作符前插，直到遇到右括号
/**
	* @param nifixExpression : 中缀表达式
	* @return: String
	* @description: 中缀转前缀
	*/
public String nifixToPrefix(String[] nifixExpression) {
  // 存储表达式拼接的结果
  StringBuilder sb = new StringBuilder();
  // 存储界限符和运算符的栈
  Stack<Character> stack = new Stack<>();
  char top;
  for (int i = nifixExpression.length - 1; i >= 0; i--) {
    String temp = nifixExpression[i];
    // 如果长度是1,可能是：1位数字,界限符,操作符
    if(temp.length() == 1) {
      char item = temp.charAt(0);
      // 如果是操作符
      if(isOp(item)) {
        if(stack.isEmpty()) stack.push(item);
        else {
          while(!stack.isEmpty() && (top = stack.peek())!=')' && cmp(item, top) < 0)
            sb.insert(0, top);
          stack.push(item);
        }
      } else if(item == ')' || item == '(') {
        // 如果是左括号,不断出栈直到遇到右括号
        if(item == '(') {
          while(!stack.isEmpty() && (top = stack.pop())!=')')
            sb.insert(0, top);
        } else stack.push(')');
      } else {
        // 如果是1位数字
        sb.insert(0, item);
      }
    } else {
      // 如果长度不为1,说明是多位的数字,进行前插
      sb.insert(0, temp);
    }
  }
  while(!stack.isEmpty()) sb.insert(0, stack.pop());
  return sb.toString();
}

2.2.2 前缀——>中缀表达式

该前缀表达式的运算符从右向左依次生效，故前缀转中缀规则：

  • 从右向左遍历
  • 将操作数压栈
  • 遇到操作符，则将栈中最上面的两个元素弹出比如A和B，进行拼接。最后将拼接结果压栈
  • 整个数组遍历结束后，栈中就是中缀结果
/**
	* @param prefixExpression: 前缀表达式
	* @return: String
	* @description: 后缀转中缀
	*/
public String PrefixToNifix(String[] prefixExpression) {
  if (prefixExpression.length == 0) return null;
  // 栈
  Stack<String> stack = new Stack<>();
  String top1, top2;
  for (int i = prefixExpression.length - 1; i >= 0; i--) {
    String item = prefixExpression[i];
    if (item.length() == 1 && isOp(item.charAt(0))) {
      // 如果是操作符,则取出栈中最上面两个元素进行合并
      top1 = stack.pop();
      top2 = stack.pop();
      // 合并后重新压栈
      stack.push('(' + top1 + item + top2 + ')');
    } else {
      // 如果是操作数直接压栈
      stack.push(item + "");
    }
  }
  // 此时栈顶元素就是中缀表达式
  String res = stack.pop();
  // 处理多余的括号
  return res.substring(1, res.length() - 1);
}

2.2.3 中缀——>后缀表达式

遵循左优先，倘若一个中缀表达式左边的运算符能够先运算，该左边的运算符可以运算

规则：

  • 遇到操作数。直接加入后缀表达式
  • 遇到界限符。比如()[]{}，遇到左半部分入栈，遇到右半部分依次弹出栈内运算符并加入后缀表达式，直到弹出对应的左半部分界限符
  • 遇到运算符。依次弹出栈中优先级高于或等于当前运算符的所有运算符，并加入后缀表达式。若遇到左半部分界限符或者栈空，则停止。之后再把当前运算符入栈。
  • 将栈中剩余元素弹出加入表达式
/**
	* @param nifixExpression : 中缀表达式
	* @return: String
	* @description: 中缀转后缀
	*/
public String nifixToPostfix(String[] nifixExpression) {
  StringBuilder sb = new StringBuilder();
  // 存储界限符和运算符的栈
  Stack<Character> stack = new Stack<>();
  char top, item;
  for (String s : nifixExpression) {
    // 每个s判断长度,如果是1,则可能为：一位的数字,操作符或者界限符
    if (s.length() == 1) {
      item = s.charAt(0);
      if (isOp(item)) {
        if(stack.isEmpty()) stack.push(item);
        else {
          // 如果遇到运算符,将优先级高于或等于item的运算符出栈加入表达式,直到栈空或遇到左半部分括号
          while(!stack.isEmpty() && (top = stack.peek())!= '(' && cmp(item, top) <= 0) sb.append(stack.pop());
          stack.push(item);
        }
      } else if (item == ')' || item == '(') {
        // 如果是界限符的右半部分
        if (item == ')') {
          // 不断出栈加入表达式中,直到遇到右半部分的括号
          while (!stack.isEmpty() && (top = stack.pop()) != '(')
            sb.append(top);
        } else
          stack.push(item);
      } else {
        sb.append(s);
      }
    } else {
      // 长度大于1说明一定是多位数字操作数
      sb.append(s);
    }
  }
  while(!stack.isEmpty()) sb.append(stack.pop());
  return sb.toString();
}

2.2.4 后缀——>中缀表达式

该后缀表达式的运算符从左向右依次生效，故后缀转中缀规则：

  • 从左向右遍历
  • 将操作数压栈
  • 遇到操作符，则将栈中最上面的两个元素弹出比如A和B，进行拼接。需要注意拼接时，先弹出来的要放到操作符的后面，比如B*A。最后将拼接结果压栈
  • 整个数组遍历结束后，栈中就是中缀结果
/**
	* @param postfixExpression: 后缀表达式
	* @return: String
	* @description: 后缀转中缀
	*/
public String PostfixToNifix(String[] postfixExpression) {
  if(postfixExpression.length == 0) return null;
  Stack<String> stack = new Stack<>();
  String top1, top2;
  for (String item : postfixExpression) {
    if (item.length() == 1 && isOp(item.charAt(0))) {
      // 如果是操作符,则取出栈中最上面两个元素进行合并
      top1 = stack.pop();
      top2 = stack.pop();
      // 合并后重新压栈
      stack.push('(' + top2 + item + top1 + ')');
    } else {
      // 如果是操作数直接压栈
      stack.push(item + "");
    }
  }
  // 此时栈顶元素就是中缀表达式
  String res = stack.pop();
  // 处理多余的括号
  return res.substring(1, res.length() - 1);
}

3.2.1 先序遍历（非递归）

/**
  * @param root: 根节点
  * @return: void
  * @description: 先序遍历,非递归
  */
public void preOrderNonRecursion(TreeNode root) {
  /**
	 * 规则：
	 * 1. 先将根节点入栈
	 * 2. 栈不为空, 出栈, 访问, 将出栈元素的右孩子、左孩子入栈
	 * 3. 循环第二步
	 */
  if (root == null) return;
  StringBuilder res = new StringBuilder(); // 存储访问结果
  TreeNode p = root; // 指针
  Stack<TreeNode> stack = new Stack<>(); // 栈
  stack.push(p); // 根节点入栈
  while (!stack.isEmpty()) {
    // 元素出栈
    p = stack.pop();
    // 访问出栈元素
    res.append(p.getElem());
    // 右孩子存在,入栈
    if (p.getrChild() != null)
      stack.push(p.getrChild());
    // 左孩子存在,入栈
    if (p.getlChild() != null)
      stack.push(p.getlChild());
  }
  System.out.println(res.toString());
}

3.2.2 中序遍历（非递归）

/**
  * @param root: 根节点
  * @return: void
  * @description: 中序遍历,非递归
  */
public void inOrderNonRecursion(TreeNode root) {
  /**
	 * 规则：
	 * 1. 先将根节点入栈
	 * 2. 如果该节点有左孩子,循环将所有左孩子入栈
	 * 3. 出栈,访问,如果出栈元素有右孩子,压栈
	 */
  if (root == null) return;
  StringBuilder res = new StringBuilder(); // 存储访问结果
  TreeNode p = root; // 指针
  Stack<TreeNode> stack = new Stack<>(); // 栈
  while (!stack.isEmpty() || p != null) {
    // 将节点的所有左孩子入栈
    while (p != null) {
      stack.push(p);
      p = p.getlChild();
    }
    if (!stack.isEmpty()) {
      // 出栈, 访问, 若有右孩子, 则入栈
      p = stack.pop();
      res.append(p.getElem());
      p = p.getrChild();
    }
  }
  System.out.println(res.toString());
}

3.2.3 后序遍历（非递归|双栈）

/**
  * @param root: 根节点
  * @return: void
  * @description: 后序遍历,非递归,双栈
  */
public void postOrderNonRecursion(TreeNode root) {
  /**
	 * 规则：
	 * 1. 先按照根-右-左的顺序入栈
	 * 2.等所有节点入栈结束后, 一个个出栈
	 */
  if (root == null) return;
  StringBuilder res = new StringBuilder(); // 存储访问结果
  TreeNode p = root; // 指针
  Stack<TreeNode> stack1 = new Stack<>(); // 栈
  Stack<TreeNode> stack2 = new Stack<>(); // 栈
  stack1.push(p);
  while (!stack1.isEmpty()) {
    // 元素出栈
    p = stack1.pop();
    // 访问出栈元素
    stack2.push(p);
    // 左孩子存在,入栈
    if (p.getlChild() != null)
      stack1.push(p.getlChild());
    // 右孩子存在,入栈
    if (p.getrChild() != null)
      stack1.push(p.getrChild());
  }
  while(!stack2.isEmpty()) {
    p = stack2.pop();
    // 访问出栈元素
    res.append(p.getElem());
  }
  System.out.println(res.toString());
}

3.2.4 后序遍历（非递归|单栈）

class MyEntry<K, V> implements Map.Entry<K,V> {
  private final K key;
  private V value;
  MyEntry(K key, V value) {
    this.key = key;
    this.value = value;
  }
  @Override
  public K getKey() {
    return key;
  }
  @Override
  public V getValue() {
    return value;
  }
  @Override
  public V setValue(V value) {
    V old = this.value;
    this.value = value;
    return old;
  }
}
  • 实现Map.Entry接口，重写方法
  • stack中存储的是一个键值对，key是树的节点，value是该节点的右孩子是否被访问，是boolean类型
/**
	* @param root: 根节点
	* @return: void
	* @description: 后序遍历,非递归,一个栈
	*/
public void postOrderNonRecursion(TreeNode root) {
  if (root == null) return;
  StringBuilder res = new StringBuilder(); // 存储访问结果
  TreeNode bt = root; // 指针
  Stack<MyEntry<TreeNode, Boolean>> stack = new Stack<>();
  MyEntry<TreeNode, Boolean> stackNode;
  while(bt != null || !stack.isEmpty()) {
    // 将该bt指向的节点以及所有的左孩子入栈
    while(bt != null) {
      // false表示该节点的右孩子没有被访问过
      stackNode = new MyEntry<>(bt, false);
      stack.push(stackNode);
      bt = bt.getlChild();
    }
    // 栈不空,并且栈顶元素的右孩子被访问过
    while(!stack.isEmpty() && stack.peek().getValue())
      // 出栈,加入遍历结果中
      res.append(stack.pop().getKey().getElem());
    if(!stack.isEmpty()) {
      // 令栈顶的右孩子被访问
      stack.peek().setValue(true);
      // bt指向栈顶节点的右孩子
      bt = stack.peek().getKey().getrChild();
    }
  }
  System.out.println(res.toString());
}

  int i = 1;

    if(i < arr.length) {

      node.setrChild(temp);
      queue.add(temp);
    }
  }
  queue.clear();
  return root;
}

			else if (node.getrChild() != null)

			else
				tag = true;
		}

	}

3.10 已知先序和中序构建二叉树

/**
	* @param inOrder:  中序遍历数组
	* @param preOrder: 先序遍历数组
	* @param li:       中序遍历数组的左边界
	* @param ri:       中序遍历数组的右边界
	* @param lp:       先序遍历数组的左边界
	* @param rp:       先序遍历数组的右边界
	* @return: tree.TreeNode
	* @description: 根据中序和先序序列创建树, 递归
	*/
public TreeNode createTreeByInOrderAndPreOrder(int[] inOrder, int[] preOrder, int li, int ri, int lp, int rp) {
  // 创建根节点
  TreeNode root = new TreeNode();
  root.setElem(preOrder[lp]);
  // 将根节点到中序遍历序列去遍历定位
  int i;
  for (i = li; i < inOrder.length && inOrder[i] != root.getElem(); i++) ;
  // 此时,i的这个位置的左边就是左子树序列,右边就是右子树序列
  // 左子树节点个数：i-li; 右子树节点个数ri-i
  // 左子树的节点在先序遍历数组中的下标范围是：[lp+1,lp+i-li]
  // 右子树的节点在先序遍历数组中的下标范围是：[lp+i-li+1,rp]
  if (i == li) root.setlChild(null);
  else root.setlChild(createTreeByInOrderAndPreOrder(inOrder, preOrder, li, i - 1, lp + 1, lp + i - li));
  if (i == ri) root.setrChild(null);
  else root.setrChild(createTreeByInOrderAndPreOrder(inOrder, preOrder, i + 1, ri, lp + i - li + 1, rp));
  return root;
}

public TreeNode createTreeByInOrderAndPostOrder(int[] inOrder, int[] postOrder, int li, int ri, int lp, int rp) {

  TreeNode root = new TreeNode();

  // 左子树节点个数：i-li; 右子树节点个数ri-i

/**
	* @param arr: 数组
	* @return: TreeNode
	* @description: 构建二叉排序树
	*/
public TreeNode createBST(int[] arr) {
  if (arr.length == 0) return null;
  // 创建根节点
  TreeNode root = new TreeNode();
  root.setElem(arr[0]);
  Stack<TreeNode> stack = new Stack<>();
  // 根节点入栈
  stack.push(root);
  TreeNode temp, top;
  for (int i = 1; i < arr.length; i++) {
    temp = new TreeNode();
    temp.setElem(arr[i]);
    while (!stack.isEmpty()) {
      top = stack.peek();
      // 如果栈顶元素关键字的值<当前要插入的节点
      if (top.getElem() < temp.getElem()) {
        if (top.getrChild() == null) {
          top.setrChild(temp);
          break;
        } else {
          stack.push(top.getrChild());
        }
      } else {
        // 如果栈顶元素关键字的值>当前要插入的节点
        if (top.getlChild() == null) {
          top.setlChild(temp);
          break;
        } else {
          stack.push(top.getlChild());
        }
      }
    }
  }
  return root;
}

4.1 AVL树

平衡二叉树（AVL Tree），要求每个节点的左右子树高度差在[-1,1]取值，下图列举了在构建平衡二叉树过程中出现的四大类不平衡情况，以及对应类型的说明。

插入节点后，原本平衡的二叉树变的不平衡，假设插入位置是c，距离插入位置c最近的不平衡节点为a，对上面四种类型做出解释：

  • LL：c在a的左子树的左子树
  • RR：c在a的右子树的右子树
  • LR：c在a的左子树的右子树
  • RL：c在a的右子树的左子树

判断是哪种类型的不平衡，是根据a，c两个节点相对位置而言的，而类型后面的文字则是表示调整这种类型不平衡需要旋转的方向。在下面方法命名中，以旋转方向命名。

4.2 构建一颗平衡二叉树

public class AVLTree {
  // 全局变量,记录插入节点后,整棵树的高度是否增加
  boolean taller = false;

  public AVLTreeNode createAVLTree(int[] arr) {
    // 创建根节点
    AVLTreeNode root = null;
    // 不断插入节点
    for (int j : arr) {
      // 如果该元素在树中找不到,则进行插入
      if (!findInAVL(root, j))
        root = insertNode(root, j);
    }
    // 返回根节点
    return root;
  }

  private AVLTreeNode insertNode(AVLTreeNode root, int elem) {
    // 如果树空,则直接插入
    if (root == null) {
      root = new AVLTreeNode();
      root.data = elem;
      // 全局高度会增加
      taller = true;
    } else if (root.data < elem) {
      AVLTreeNode temp = insertNode(root.rightChild, elem);
      if(root.rightChild != temp) root.rightChild = temp;
      // 如果此时insertNode函数传过来的root节点值 < elem,说明elem应该在root的右子树进行插入
      // root指向的是离插入节点Node最近的失衡节点或者Node节点的父亲节点
      if (taller) {
        // 现在已经明确往右子树添加一个节点Node,此时的taller为true只是说明添加新节点后可能会使
        // 整个树的高度增加,实际上增不增加,需要看Node的父亲节点的平衡因子,也就是后面的switch干的事情
        switch (root.bf) {
            // root的bf是0,说明添加Node之前左子树高,所以右子树添加Node后,不会影响树的平衡
            // 并且还能让root的bf变成0,并且可以知道此时的整个树并没有增高
          case 1:
            root.bf = 0;
            taller = false;
            break;
            // 倘若root节点的bf为0,说明root节点左右之树原先一样高,那么在其右子树添加一个节点必然会导致右子树变高
          case 0:
            root.bf = -1;
            taller = true;
            break;
            // 倘若root节点bf为-1,说明未添加Node节点之前,root的右子树就高,此时又添加一个节点,会导致平衡破坏
          case -1:
            // 因为是右子树的插入导致的失衡,调用rightBalance方法,此时传递的root节点指向的是距离新插入的Node最近的失衡节点
            root = rightBalance(root);
            taller = false;
            break;
          default:
            throw new IllegalStateException("Unexpected value: " + root.bf);
        }
      }
    } else {
      AVLTreeNode temp = insertNode(root.leftChild, elem);
      if(root.leftChild != temp) root.leftChild = temp;
      switch (root.bf) {
        case 1:
          root = leftBalance(root);
          taller = false;
          break;
        case 0:
          root.bf = 1;
          taller = true;
          break;
        case -1:
          root.bf = 0;
          taller = false;
          break;
      }
    }
    // 返回节点
    return root;
  }

  /**
	 * @param root:
	 * @return: null
	 * @description: 进行平衡调整
	 */
  private AVLTreeNode rightBalance(AVLTreeNode root) {
    // root为Node节点的祖先节点,temp的父亲节点
    // temp指向root的右子树,temp指向插入节点Node的父亲节点
    AVLTreeNode temp = root.rightChild, rd;
    switch (temp.bf) {
      case 1:
        rd = temp.leftChild;
        switch (rd.bf) {
          case 1:
            root.bf = 0;
            temp.bf = -1;
            break;
          case 0:
            root.bf = temp.bf = 0;
            break;
          case -1:
            root.bf = 0;
            temp.bf = 1;
            break;
        }
        rd.bf = 0;
        root.rightChild = R_Rotate(temp);
        root = L_Rotate(root);
        break;
      case -1:
        root.bf = temp.bf = 0;
        // 左旋后,根节点发生变化,进行更新
        root = L_Rotate(root);
        break;
    }
    return root;
  }

  private AVLTreeNode leftBalance(AVLTreeNode root) {
    // root为Node节点的祖先节点,temp的父亲节点
    // temp指向root的右子树,temp指向插入节点Node的父亲节点
    AVLTreeNode temp = root.leftChild, rd;
    switch (temp.bf) {
      case -1:
        rd = temp.rightChild;
        switch (rd.bf) {
          case 1:
            root.bf = -1;
            temp.bf = 0;
            break;
          case 0:
            root.bf = temp.bf = 0;
            break;
          case -1:
            root.bf = 0;
            temp.bf = 1;
            break;
        }
        rd.bf = 0;
        root.leftChild = L_Rotate(temp);
        root = R_Rotate(root);
        break;
      case 1:
        root.bf = temp.bf = 0;
        root = R_Rotate(root);
        break;
    }
    return root;
  }

  /**
	 * @param root: 不平衡的树的根节点
	 * @description: LL类型不平衡的右旋处理
	 */
  private AVLTreeNode R_Rotate(AVLTreeNode root) {
    AVLTreeNode temp = root.leftChild;
    root.leftChild = temp.rightChild;
    temp.rightChild = root;
    // 返回调整过后的新的根节点
    return temp;
  }

  /**
	 * @param root: 不平衡的树的根节点
	 * @description: RR类型不平衡的左旋处理
	 */
  private AVLTreeNode L_Rotate(AVLTreeNode root) {
    AVLTreeNode temp = root.rightChild;
    root.rightChild = temp.leftChild;
    temp.leftChild = root;
    // 返回调整过后的新的根节点
    return temp;
  }

  /**
	 * @param root: 树根节点
	 * @param elem: 需要被查找的元素
	 * @description: 判断该二叉排序树中是否有elem这个节点
	 */
  private boolean findInAVL(AVLTreeNode root, int elem) {
    if (root == null) return false;
    if (root.data == elem) return true;
    else if (root.data > elem) {
      return findInAVL(root.leftChild, elem);
    } else {
      return findInAVL(root.rightChild, elem);
    }
  }
}
